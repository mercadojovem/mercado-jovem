// supabase/functions/lancar-pregao/index.ts // Edge Function — Lançamento de Pregão em Lote // Autentica o admin, cria o pregão e insere todas as operações atomicamente. import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'; import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, content-type', }; interface Operacao { comprovante_num: string | null; empresa_compradora: string; empresa_vendedora: string; qtd_cotas: number; valor_por_cota: number; valor_total: number; observacao: string | null; } interface Pregao { numero: number; data: string; observacao?: string; } serve(async (req: Request) => { if (req.method === 'OPTIONS') return new Response('ok', { headers: CORS }); if (req.method !== 'POST') { return new Response(JSON.stringify({ error: 'Método não permitido' }), { status: 405, headers: { ...CORS, 'Content-Type': 'application/json' } }); } // ── Autenticação: valida o token do admin via auth-admin ── const authHeader = req.headers.get('Authorization') || ''; const token = authHeader.replace('Bearer ', '').trim(); if (!token) { return json({ error: 'Token de autorização ausente.' }, 401); } // Reutiliza a mesma verificação do auth-admin // Aqui assumimos que o token é o hash SHA-256 já validado guardado na sessão // Para maior segurança, verificar contra Supabase ou um segredo compartilhado const ADMIN_TOKEN = Deno.env.get('ADMIN_SESSION_SECRET') || ''; if (token !== ADMIN_TOKEN && ADMIN_TOKEN !== '') { return json({ error: 'Não autorizado.' }, 401); } // ── Parse do body ── let body: { pregao: Pregao; operacoes: Operacao[] }; try { body = await req.json(); } catch { return json({ error: 'JSON inválido.' }, 400); } const { pregao, operacoes } = body; if (!pregao?.numero || !pregao?.data) { return json({ error: 'Número e data do pregão são obrigatórios.' }, 400); } if (!Array.isArray(operacoes) || operacoes.length === 0) { return json({ error: 'Nenhuma operação fornecida.' }, 400); } // ── Validações ── for (let i = 0; i < operacoes.length; i++) { const op = operacoes[i]; if (!op.empresa_compradora || !op.empresa_vendedora) { return json({ error: `Operação ${i + 1}: empresas não informadas.` }, 400); } if (op.empresa_compradora === op.empresa_vendedora) { return json({ error: `Operação ${i + 1}: compradora e vendedora não podem ser iguais.` }, 400); } if (!op.qtd_cotas || op.qtd_cotas <= 0) { return json({ error: `Operação ${i + 1}: quantidade de cotas inválida.` }, 400); } if (!op.valor_por_cota || op.valor_por_cota <= 0) { return json({ error: `Operação ${i + 1}: valor por cota inválido.` }, 400); } } // ── Supabase com service_role ── const supabase = createClient( Deno.env.get('SB_URL')!, Deno.env.get('SB_SERVICE_KEY')! ); // ── Inserir pregão ── const { data: pregaoData, error: errPregao } = await supabase .from('pregoes') .insert({ numero: pregao.numero, data: pregao.data, observacao: pregao.observacao || null, status: 'aberto' }) .select('id') .single(); if (errPregao) { // Se pregão com esse número já existe, busca o id existente if (errPregao.code === '23505') { return json({ error: `Pregão nº ${pregao.numero} já foi lançado anteriormente.` }, 409); } return json({ error: 'Erro ao criar pregão: ' + errPregao.message }, 500); } const pregaoId = pregaoData.id; // ── Inserir operações em lote ── const opsParaInserir = operacoes.map(op => ({ pregao_id: pregaoId, comprovante_num: op.comprovante_num, empresa_compradora: op.empresa_compradora, empresa_vendedora: op.empresa_vendedora, qtd_cotas: op.qtd_cotas, valor_por_cota: op.valor_por_cota, observacao: op.observacao, lancado_por: 'BancoCentral' })); const { error: errOps } = await supabase .from('operacoes_pregao') .insert(opsParaInserir); if (errOps) { // Rollback manual: remove o pregão criado await supabase.from('pregoes').delete().eq('id', pregaoId); return json({ error: 'Erro ao inserir operações: ' + errOps.message }, 500); } // ── Atualiza saldos das empresas ── // Para cada operação: crédito no vendedor, débito no comprador (em JCV) for (const op of operacoes) { const valorTotal = op.qtd_cotas * op.valor_por_cota; // Debitar comprador await supabase.rpc('ajustar_saldo_empresa', { p_nome: op.empresa_compradora, p_delta: -valorTotal }); // Creditar vendedor await supabase.rpc('ajustar_saldo_empresa', { p_nome: op.empresa_vendedora, p_delta: +valorTotal }); } return json({ ok: true, pregao_id: pregaoId, operacoes_inseridas: operacoes.length, message: `Pregão nº ${pregao.numero} lançado com sucesso.` }, 201); }); function json(body: unknown, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...CORS, 'Content-Type': 'application/json' } }); }
